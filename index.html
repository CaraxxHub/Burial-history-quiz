        // Utility function to shuffle an array

        function shuffleArray(array) {

            for (let i = array.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [array[i], array[j]] = [array[j], array[i]];

            }

            return array;

        }

            const checkAnswerButton = document.getElementById('check-answer-button');

            const showHintButton = document.getElementById('show-hint-button');

            const nextQuestionButton = document.getElementById('next-question-button');

            const restartQuizButtonTop = document.getElementById('restart-quiz-button-top');


            optionsContainer.addEventListener('click', (event) => {

                const targetButton = event.target.closest('.option-button');

                if (targetButton && !checkAnswerButton.disabled) { // Only allow selection if not already checked

                    // Deselect previous

                    document.querySelectorAll('.option-button').forEach(btn => {

                        btn.classList.remove('bg-blue-200', 'border-blue-500');

                    });

                    // Select new

                    targetButton.classList.add('bg-blue-200', 'border-blue-500');

                    selectedAnswerIndex = parseInt(targetButton.dataset.index);

                    updateCheckAnswerButtonState();

                }

            });


            checkAnswerButton.addEventListener('click', checkAnswer);

            showHintButton.addEventListener('click', showHint);

            nextQuestionButton.addEventListener('click', handleNextQuestion);

            restartQuizButtonTop.addEventListener('click', handleRestartQuiz);

        }


        function updateCheckAnswerButtonState() {

            const checkAnswerButton = document.getElementById('check-answer-button');

            if (selectedAnswerIndex !== null) {

                checkAnswerButton.disabled = false;

                checkAnswerButton.classList.remove('opacity-50', 'cursor-not-allowed');

            } else {

                checkAnswerButton.disabled = true;

                checkAnswerButton.classList.add('opacity-50', 'cursor-not-allowed');

            }

        }


        function checkAnswer() {

            if (selectedAnswerIndex === null) {

                showMessageBox("Please select an answer before checking.");

                return;

            }


            const correctAnswer = shuffledOptions.find(option => option.isCorrect);

            const selectedOption = shuffledOptions[selectedAnswerIndex];

            const feedbackContainer = document.getElementById('feedback-container');

            const optionButtons = document.querySelectorAll('.option-button');

            const checkAnswerButton = document.getElementById('check-answer-button');

            const nextQuestionButton = document.getElementById('next-question-button');

            const showHintButton = document.getElementById('show-hint-button');


            checkAnswerButton.disabled = true; // Disable checking after an attempt

            checkAnswerButton.classList.add('opacity-50', 'cursor-not-allowed');

            showHintButton.classList.add('hidden'); // Hide hint button after checking


            // Disable all option buttons

            optionButtons.forEach(btn => {

                btn.disabled = true;

                btn.classList.remove('hover:bg-blue-50', 'focus:ring-2'); // Remove hover/focus styles

            });


            feedbackContainer.classList.remove('hidden');


            if (selectedOption.isCorrect) {

                score++;

                feedbackContainer.classList.remove('bg-red-100', 'text-red-800', 'border-red-300');

                feedbackContainer.classList.add('bg-green-100', 'text-green-800', 'border-green-300');

                feedbackContainer.innerHTML = `<p class="font-bold">Correct!</p><p>${selectedOption.rationale}</p>`;

            } else {

                feedbackContainer.classList.remove('bg-green-100', 'text-green-800', 'border-green-300');

                feedbackContainer.classList.add('bg-red-100', 'text-red-800', 'border-red-300');

                feedbackContainer.innerHTML = `<p class="font-bold">Incorrect.</p><p>${selectedOption.rationale}</p>`;

            }


            // Highlight correct answer

            optionButtons.forEach((btn, index) => {

                if (shuffledOptions[index].isCorrect) {

                    btn.classList.add('bg-green-300', 'border-green-600', 'font-semibold');

                } else if (index === selectedAnswerIndex && !selectedOption.isCorrect) {

                    btn.classList.add('bg-red-300', 'border-red-600', 'font-semibold');

                }

            });


            nextQuestionButton.classList.remove('hidden'); // Show next question button

        }


        function showHint() {

            const hintDisplay = document.getElementById('hint-display');

            const questionData = quizQuestions[currentQuestionIndex];

            if (questionData.hint) {

                hintDisplay.textContent = `Hint: ${questionData.hint}`;

                hintDisplay.classList.remove('hidden');

            } else {

                showMessageBox("No hint available for this question.");

            }

        }


        function handleNextQuestion() {

            currentQuestionIndex++;

            renderQuestion();

        }


        function renderResults() {

            quizContent.innerHTML = `

                <div class="text-center">

                    <h2 class="text-2xl font-bold text-gray-900 mb-4">Quiz Completed!</h2>

                    <p class="text-xl text-gray-700 mb-6">You scored <span class="font-extrabold text-blue-600">${score}</span> out of <span class="font-extrabold text-blue-600">${quizQuestions.length}</span> questions.</p>

                    <button id="restart-quiz-button-bottom" class="px-6 py-3 bg-blue-600 text-white font-medium rounded-md hover:bg-blue-700 transition-colors duration-200">

                        Restart Quiz

                    </button>

                </div>

            `;

            document.getElementById('restart-quiz-button-bottom').addEventListener('click', handleRestartQuiz);

        }


        function handleRestartQuiz() {

            currentQuestionIndex = 0;

            score = 0;

            selectedAnswerIndex = null;

            shuffledOptions = [];

            quizStarted = false; // Reset quiz started state

            startQuiz(); // Re-start the quiz

        }


        function startQuiz() {

            // Initial screen or directly render the first question

            if (!quizStarted) {

                quizContent.innerHTML = `

                    <div class="text-center">

                        <p class="text-lg text-gray-700 mb-6">Test your knowledge of Burial History with this interactive quiz!</p>

                        <button id="start-button" class="px-8 py-4 bg-blue-600 text-white font-bold text-xl rounded-lg shadow-lg hover:bg-blue-700 transition-colors duration-200 transform hover:scale-105">

                            Start Quiz

                        </button>

                    </div>

                `;

                document.getElementById('start-button').addEventListener('click', () => {

                    quizStarted = true;

                    renderQuestion();

                });

            } else {

                renderQuestion(); // This branch is for restarting after completion

            }

        }


        // Initialize the quiz

        startQuiz();


    </script>
